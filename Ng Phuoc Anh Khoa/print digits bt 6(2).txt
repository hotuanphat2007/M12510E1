#include <stdio.h>
int main ()
{
    int number,count = 0;
    printf("Enter number : ");scanf("%d",&number);
    if (number <0)
    {
        number = -number;
    }
    if (number == 0)
    {
        count = -1;
    }
    else 
    {
        while (number > 0 )
        {
            number = number /10;
            // count ++ ; 
            count = count + 1;
        }
    }
    printf("the number has %d digits \n",count);

    return 0;   
}
